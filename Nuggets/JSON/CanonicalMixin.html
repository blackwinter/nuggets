def _nuggets_json_canonical(obj)
  case obj
    when ::Hash
      obj.class.new.tap { |res|
        obj.keys.sort.each { |k| res[k] = send(__method__, obj[k]) } }
    when ::Array
      obj.map { |v| send(__method__, v) }.sort_by(&:to_s)
    else
      obj
  end
end
