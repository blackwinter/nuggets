def convert(string)
  string.gsub(COLOR_RE) { format($1.split(DELIMITER).uniq) }
end

def format(attributes)
  "#{clear(attributes)}#{style(attributes) if attributes.any?}"
end

def style(attributes)
  STYLE % ATTRIBUTES.values_at(*attributes).join('; ')
end
