def covariance
  sx, sy, sp = 0.0, 0.0, 0.0

  return sx if empty?

  target = first.respond_to?(:to_ary) ? self :
    ::Array.new(size) { |i| i + 1 }.zip(self)

  target.each { |x, y|
    sx += x
    sy += y
    sp += x * y
  }

  (sp - sx * sy / size) / size
end

def variance
  sx, sq = 0.0, 0.0

  return sx if empty?

  each { |x|
    x = yield x if block_given?

    sx += x
    sq += x ** 2
  }

  (sq - sx ** 2 / size) / size
end
