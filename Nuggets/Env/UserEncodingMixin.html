    Finds the user's selected encoding.

def user_encoding(default = 'UTF-8')
  self['ENCODING'] || begin
    lang = self['LANG']
    lang[/\.(.*)/, 1] if lang
  end || if defined?(::Win32::Console)
    "CP#{::Win32::Console.InputCP}"
  elsif ::File::ALT_SEPARATOR
    cp = %x{chcp}[/:\s*(.*?)\./, 1]
    "CP#{cp}" if cp
  end || default
end
