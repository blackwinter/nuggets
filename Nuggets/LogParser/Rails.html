def parse_line(line, entry = {})
  ITEMS.each { |key, item|
    if md = item[:re].match(line)
      if key == :processing
        yield if block_given?
        entry[:pid], entry[:host] = md[1], md[2]
      end

      item[:proc][entry, md]

      break
    end
  }

  entry
end
